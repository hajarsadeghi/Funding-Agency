def get_top_n_labels(df, n):
  # create a group using groupby
  g_df = df.groupby('Cluster_Labels')
  # size of group to count observations
  g_df = g_df.size()
  # make a column name
  g_df = g_df.reset_index(name='Observation').sort_values(by='Observation', ascending=False)
  final_df = g_df.reset_index(drop=True)

  return final_df.head(n)

def plot_bar_chart(top_20, cluster_num):
  plt.figure(figsize=(10, 8))

  plt.barh(top_20['Cluster_Labels'].astype(str), top_20['Observation'], color='skyblue')
  plt.xlabel('Count')
  plt.ylabel('Labels')
  plt.title(f'Top 20 Label Counts for cluster: {cluster_num}')
  plt.show()
